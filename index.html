async function printMemoHistory() { const sel = document.getElementById('memo-circle'); const cn = sel && sel.value ? sel.options[sel.selectedIndex].text : 'جميع الحلقات'; const circleId = sel?.value || null; // جلب IDs الطلاب المستهدفين let sids = []; if (circleId) { const { data: cs } = await sb.from('students').select('id').eq('circle_id', circleId); sids = cs?.map(s=>s.id) || []; } else if (CP.role === 'teacher') { const { data: mc } = await sb.from('circles').select('id').eq('teacher_id', CU.id); const cids = mc?.map(c=>c.id) || []; if (cids.length) { const { data: cs } = await sb.from('students').select('id').in('circle_id', cids); sids = cs?.map(s=>s.id) || []; } } // جلب الحفظ والمراجعة مع اسم الطالب مباشرة let hQ = sb.from('memorization_log') .select('student_id,surah_from,ayah_from,surah_to,ayah_to,verses_count,mastery,date,students(id,full_name)') .order('date',{ascending:false}).limit(500); let mQ = sb.from('review_log') .select('student_id,surah_from,ayah_from,surah_to,ayah_to,verses_count,mastery,date,students(id,full_name)') .order('date',{ascending:false}).limit(500); if (sids.length) { hQ = hQ.in('student_id', sids); mQ = mQ.in('student_id', sids); } const [{ data: hData }, { data: mData }] = await Promise.all([hQ, mQ]); // بناء map بالاسم من البيانات نفسها - مع جمع جميع السجلات const studentsMap = {}; // معالجة الحفظ - جمع جميع السجلات (hData||[]).forEach(h => { const sid = h.student_id; if (!studentsMap[sid]) { studentsMap[sid] = { name: h.students?.full_name || '—', hifz: [], mur: [] }; } studentsMap[sid].hifz.push({ surah_from: h.surah_from, ayah_from: h.ayah_from || 1, surah_to: h.surah_to, ayah_to: h.ayah_to || 1, verses_count: h.verses_count || 0, mastery: h.mastery, date: h.date }); }); // معالجة المراجعة - جمع جميع السجلات (mData||[]).forEach(m => { const sid = m.student_id; if (!studentsMap[sid]) { studentsMap[sid] = { name: m.students?.full_name || '—', hifz: [], mur: [] }; } studentsMap[sid].mur.push({ surah_from: m.surah_from, ayah_from: m.ayah_from || 1, surah_to: m.surah_to, ayah_to: m.ayah_to || 1, verses_count: m.verses_count || 0, mastery: m.mastery, date: m.date }); }); const activeList = Object.values(studentsMap).sort((a,b) => a.name.localeCompare(b.name,'ar')); if (!activeList.length) { alert('لا توجد سجلات حفظ أو مراجعة'); return; } // ترجمة التقديرات const mAr = { excellent: 'ممتاز', good: 'جيد', needs_review: 'يحتاج مراجعة', weak: 'ضعيف' }; const mBg = { excellent: '#d4f0dc', good: '#d0e8fb', needs_review: '#fff3cd', weak: '#fde0e0' }; const mCl = { excellent: '#1a6b38', good: '#14568f', needs_review: '#856404', weak: '#a3262b' }; // دالة عرض التقدير مع خلفية ملونة const gradeBadge = (mastery) => { if (!mastery || !mAr[mastery]) { return '—'; } return `${mAr[mastery]}`; }; // دالة عرض السجلات المتعددة في خلية واحدة const formatRecords = (records, type) => { if (!records || records.length === 0) return '—'; // نأخذ أحدث 3 سجلات فقط للاختصار const sorted = [...records].sort((a,b) => new Date(b.date) - new Date(a.date)); const top = sorted.slice(0, 3); return top.map((r, idx) => { const dateStr = r.date ? new Date(r.date).toLocaleDateString('ar-KW') : ''; const versesText = r.verses_count ? `${r.verses_count} آية` : ''; const masteryText = r.mastery ? gradeBadge(r.mastery) : ''; // عرض مختلف للحفظ والمراجعة if (type === 'hifz') { return `
${r.surah_from} ← ${r.surah_to} ${versesText ? ` | ${versesText}` : ''} ${dateStr ? ` | ${dateStr}` : ''}
${masteryText}
`; } else { // المراجعة return `
${r.surah_from} ← ${r.surah_to} ${dateStr ? ` | ${dateStr}` : ''}
${masteryText}
`; } }).join(''); }; // بناء الصفوف مع 6 أعمدة (الرقم، الاسم، الحفظ + تقديره، المراجعة + تقديرها) const rows = activeList.map((s, i) => { const bg = i % 2 === 0 ? '#ffffff' : '#f5f2ea'; const hifzHtml = formatRecords(s.hifz, 'hifz'); const murHtml = formatRecords(s.mur, 'mur'); return ` ${i+1} ${s.name} ${hifzHtml} ${s.hifz && s.hifz.length > 0 ? s.hifz.slice(0,3).map(r => `
${gradeBadge(r.mastery)}
` ).join('') : '—'} ${murHtml} ${s.mur && s.mur.length > 0 ? s.mur.slice(0,3).map(r => `
${gradeBadge(r.mastery)}
` ).join('') : '—'} `; }).join(''); const printTitle = `كشف الحفظ والمراجعة — ${cn} — مركز التراث لتحفيظ القرآن الكريم — فرع الجهراء`; const printDate = `تاريخ الطباعة: ${new Date().toLocaleDateString('ar-KW')} | عدد الطلاب: ${activeList.length}`; document.getElementById('print-title').textContent = printTitle; document.getElementById('print-date').textContent = printDate; document.getElementById('print-table').innerHTML = ` ${rows}
# الطالب الحفظ الجديد تقدير الحفظ المراجعة تقدير المراجعة
`; window.print(); }
